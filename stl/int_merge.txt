#include <iostream>
using namespace std;

int N;
int arr[100];
int tmp[100];

bool comp(const int& a, const int& b) {
	return a < b;
}

void merge(int s, int e) {
	if (s < e) {
		int m = (s + e) / 2;
		int l = s, r = m + 1, k = l;
		merge(l, m);
		merge(r, e);
		while (l <= m && r <= e) {
			tmp[k++] = comp(arr[l], arr[r]) ? arr[l++] : arr[r++];
		}
		while (l <= m) tmp[k++] = arr[l++];
		while (r <= e) tmp[k++] = arr[r++];
		for (int i = s; i <= e; i++) {
			arr[i] = tmp[i];
		}
	}
}

int main() {
	cin >> N;
	for (int i = 0; i < N; i++) cin >> arr[i];
	merge(0, N - 1);
	for (int i = 0; i < N+5; i++)
		cout << arr[i] << ' ';
	return 0;
}
